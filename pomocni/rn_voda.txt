                        i.rn_tip_id);
        
        
                        
        insert into rn_stavke(rn_id, brojilo_vrsta_kolone_id, vrednost)
        values (p_id, 43, i.VO_KOLICINA);
        
        insert into rn_stavke(rn_id, brojilo_vrsta_kolone_id, vrednost)
        values (p_id, 44, i.VO_CENA);
